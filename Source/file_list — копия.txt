    <h2>⚔️ Godgrave ⚔️</h2>
    
    <div class="image-container" id="imageContainer">
        <div class="loading">🖼️ Загрузка картинки...</div>
    </div>
    
    <button id="soundBtn" disabled>🔊 Слушать звук (загружается...)</button>
    <button id="reloadBtn">🔄 Перезагрузить всё</button>
    
    <div id="status" class="status"></div>

<script>
    // ========== КОНФИГУРАЦИЯ ==========
    // Путь к картинке (возьмём любой существующий спрайт)
    const IMAGE_URL = "https://raw.githubusercontent.com/JackofShadow666/GodgraveAssets/main/Build/CharacterSet/helm_horned.png";
    
    // Путь к звуку
    const SOUND_URL = "https://raw.githubusercontent.com/JackofShadow666/GodgraveAssets/main/Source/Sound/Arrow/SFX_ArrowHit_01.mp3";
    
    // ========== ЭЛЕМЕНТЫ ==========
    const imageContainer = document.getElementById('imageContainer');
    const soundBtn = document.getElementById('soundBtn');
    const reloadBtn = document.getElementById('reloadBtn');
    const statusDiv = document.getElementById('status');
    
    let audio = null;
    let isSoundLoaded = false;
    
    // ========== ФУНКЦИЯ ЛОГИРОВАНИЯ ==========
    function setStatus(message, type = 'loading') {
        statusDiv.innerHTML = message;
        statusDiv.className = `status ${type}`;
        console.log(`[${type.toUpperCase()}] ${message}`);
    }
    
    // ========== ЗАГРУЗКА КАРТИНКИ ==========
    function loadImage() {
        setStatus('🖼️ Загрузка картинки...', 'loading');
        imageContainer.innerHTML = '<div class="loading">🖼️ Загрузка картинки...</div>';
        
        const img = new Image();
        
        img.onload = () => {
            imageContainer.innerHTML = '';
            imageContainer.appendChild(img);
            setStatus('✅ Картинка загружена!', 'success');
        };
        
        img.onerror = (error) => {
            imageContainer.innerHTML = '<div class="error">❌ Не удалось загрузить картинку</div>';
            setStatus('❌ Ошибка загрузки картинки', 'error');
            console.error('Ошибка картинки:', error);
        };
        
        img.src = IMAGE_URL;
        img.alt = 'Godgrave Armor';
        img.style.maxWidth = '300px';
        img.style.maxHeight = '200px';
        img.style.borderRadius = '10px';
    }
    
    // ========== ЗАГРУЗКА ЗВУКА ==========
    function loadSound() {
        setStatus('🔊 Загрузка звука...', 'loading');
        soundBtn.disabled = true;
        soundBtn.textContent = '🔊 Загрузка звука...';
        
        // Создаём Audio элемент
        audio = new Audio();
        
        audio.addEventListener('canplaythrough', () => {
            isSoundLoaded = true;
            soundBtn.disabled = false;
            soundBtn.textContent = '🏹 Воспроизвести звук стрелы';
            setStatus('✅ Звук загружен! Нажми на кнопку', 'success');
        });
        
        audio.addEventListener('error', (e) => {
            isSoundLoaded = false;
            soundBtn.disabled = false;
            soundBtn.textContent = '❌ Ошибка загрузки звука';
            setStatus('❌ Не удалось загрузить звук (возможно файл отсутствует)', 'error');
            console.error('Ошибка звука:', e);
        });
        
        audio.src = SOUND_URL;
        audio.load();
    }
    
    // ========== ВОСПРОИЗВЕДЕНИЕ ЗВУКА ==========
    function playSound() {
        if (!isSoundLoaded || !audio) {
            setStatus('⚠️ Звук ещё не загружен, подожди...', 'error');
            return;
        }
        
        // Останавливаем если играет
        audio.currentTime = 0;
        
        // Воспроизводим
        const playPromise = audio.play();
        
        if (playPromise !== undefined) {
            playPromise.then(() => {
                setStatus('🏹 Звук стрелы!', 'success');
                setTimeout(() => {
                    if (statusDiv.innerHTML === '🏹 Звук стрелы!') {
                        setStatus('✅ Звук загружен!', 'success');
                    }
                }, 1500);
            }).catch(error => {
                setStatus('❌ Не удалось воспроизвести (нужно взаимодействие с пользователем)', 'error');
                console.error('Ошибка воспроизведения:', error);
            });
        }
    }
    
    // ========== ПЕРЕЗАГРУЗКА ВСЕГО ==========
    function reloadAll() {
        setStatus('🔄 Перезагрузка...', 'loading');
        isSoundLoaded = false;
        audio = null;
        
        // Загружаем картинку заново
        loadImage();
        
        // Загружаем звук заново
        loadSound();
    }
    
    // ========== ИНИЦИАЛИЗАЦИЯ ==========
    function init() {
        loadImage();
        loadSound();
        
        // Вешаем обработчики
        soundBtn.onclick = playSound;
        reloadBtn.onclick = reloadAll;
    }
    
    // Запускаем всё
    init();
    
    // Показываем в консоли ссылки для проверки
    console.log('📁 Ссылка на картинку:', IMAGE_URL);
    console.log('📁 Ссылка на звук:', SOUND_URL);
    console.log('💡 Если звук не работает - возможно файл отсутствует в репозитории');
    
</script>
</body>
</html>
